Script 01 — Log Analysis
SSH Brute-Force Detector with Auto-Block
Parses /var/log/auth.log, counts failed attempts per source IP, sends alerts above threshold, and optionally blocks with iptables. Production-ready with logging and dry-run mode.
ssh_guard.sh BASH 
#!/usr/bin/env bash
# ssh_guard.sh — SSH brute-force detection & response
# Usage: ./ssh_guard.sh [--dry-run] [--threshold N] [--log FILE]

set -euo pipefail   # strict mode: exit on error, unset var, pipe fail
IFS=$'\n\t'         # safer word splitting

# ── CONFIG (overridable via args) ─────────────────
AUTH_LOG="/var/log/auth.log"
THRESHOLD=10
DRY_RUN=false
LOG_FILE="/var/log/ssh_guard.log"
WHITELIST=("127.0.0.1" "10.0.0.1")  # never block these

# ── ARG PARSING ───────────────────────────────────
while [[ $# -gt 0 ]]; do
  case $1 in
    --dry-run)   DRY_RUN=true ;;
    --threshold) THRESHOLD="$2"; shift ;;
    --log)       AUTH_LOG="$2"; shift ;;
    *) echo "Unknown arg: $1" >&2; exit 1 ;;
  esac
  shift
done

# ── HELPERS ───────────────────────────────────────
log() {
  local level="$1"; shift
  printf '[%s] [%s] %s\n' "$(date -u +%FT%TZ)" "$level" "$*" | \
    tee -a "$LOG_FILE"
}

is_whitelisted() {
  local ip="$1"
  for w in "${WHITELIST[@]}"; do
    [[ "$ip" == "$w" ]] && return 0
  done
  return 1
}

block_ip() {
  local ip="$1"
  if is_whitelisted "$ip"; then
    log WARN "$ip is whitelisted — skipping block"
    return
  fi
  if [[ $DRY_RUN == "true" ]]; then
    log DRY  "WOULD block: $ip"
  else
    iptables -A INPUT -s "$ip" -j DROP
    log BLOCK "Blocked: $ip"
  fi
}

# ── MAIN ──────────────────────────────────────────
log INFO "Starting scan — threshold=$THRESHOLD, log=$AUTH_LOG"

grep "Failed password" "$AUTH_LOG" \
  | awk '{print $(NF-3)}'           \
  | grep -Eo '[0-9]{1,3}(\.[0-9]{1,3}){3}' \
  | sort | uniq -c | sort -rn \
  | while read -r count ip; do
      if [[ $count -ge $THRESHOLD ]]; then
        log ALERT "$ip → $count failed attempts"
        block_ip "$ip"
      fi
    done

log INFO "Scan complete"

Script 02 — System Monitoring
File Integrity Monitor (Lightweight HIDS)
Baseline critical system files with SHA-256, then continuously verify them. Alerts on any unauthorized modification — a minimal HIDS.
fim.sh BASH 
#!/usr/bin/env bash
# fim.sh — File Integrity Monitor
# Usage: ./fim.sh baseline | ./fim.sh check | ./fim.sh watch [interval_secs]

set -euo pipefail

WATCH_FILES=(
  "/etc/passwd" "/etc/shadow" "/etc/sudoers"
  "/etc/ssh/sshd_config" "/etc/crontab"
  "/etc/hosts" "/etc/ld.so.preload"
)
BASELINE_DB="/var/lib/fim/baseline.db"
ALERT_LOG="/var/log/fim_alerts.log"

mkdir -p "$(dirname "$BASELINE_DB")"

hash_file() {
  sha256sum "$1" 2>/dev/null | cut -d' ' -f1
}

do_baseline() {
  echo "[$(date -u +%FT%TZ)] Generating baseline..."
  true > "$BASELINE_DB"
  for f in "${WATCH_FILES[@]}"; do
    if [[ -f "$f" ]]; then
      printf '%s  %s\n' "$(hash_file "$f")" "$f" >> "$BASELINE_DB"
      echo "  ✓ $f"
    else
      echo "  ✗ NOT FOUND: $f"
    fi
  done
  echo "Baseline saved → $BASELINE_DB"
}

do_check() {
  local violations=0
  while read -r expected_hash filepath; do
    if [[ ! -f "$filepath" ]]; then
      alert "DELETED" "$filepath"; (( violations++ ))
      continue
    fi
    local current_hash; current_hash="$(hash_file "$filepath")"
    if [[ "$current_hash" != "$expected_hash" ]]; then
      alert "MODIFIED" "$filepath"
      echo "    was: $expected_hash"
      echo "    now: $current_hash"
      (( violations++ ))
    fi
  done < "$BASELINE_DB"
  [[ $violations -eq 0 ]] && echo "[OK] All files intact"
  return $violations
}

alert() {
  local msg; msg="[$(date -u +%FT%TZ)] [ALERT] $1: $2"
  echo "$msg" | tee -a "$ALERT_LOG"
  # Hook: send to SIEM, Slack webhook, email, etc.
}

case "${1:-check}" in
  baseline) do_baseline ;;
  check)    do_check ;;
  watch)
    interval="${2:-300}"
    echo "Watching every ${interval}s — Ctrl+C to stop"
    while true; do do_check; sleep "$interval"; done ;;
  *) echo "Usage: $0 baseline|check|watch [interval]"; exit 1 ;;
esac

Script 03 — Network Monitoring
Network Asset Discovery & Change Detector
Pings a subnet, grabs hostnames, baselines the inventory, and alerts when new or missing hosts appear — lightweight asset tracking without nmap.
net_sweep.sh BASH 
#!/usr/bin/env bash
# net_sweep.sh — Subnet sweep & change detection
# Usage: ./net_sweep.sh 192.168.1 [--baseline]

set -euo pipefail
SUBNET="${1:-192.168.1}"
BASELINE="/tmp/net_baseline_${SUBNET//./_}.txt"
CURRENT=$(mktemp)
trap 'rm -f "$CURRENT"' EXIT

sweep() {
  echo "[*] Sweeping $SUBNET.0/24..."
  for i in $(seq 1 254); do
    {
      ip="$SUBNET.$i"
      if ping -c1 -W1 "$ip" >/dev/null 2>&1; then
        host_name=$(dig +short -x "$ip" 2>/dev/null || echo "unknown")
        printf '%s %s\n' "$ip" "${host_name%.}" >> "$CURRENT"
      fi
    } &  # parallel pings
  done
  wait  # wait for all background jobs
  sort -t. -k4 -n "$CURRENT"
}

if [[ "${2:-}" == "--baseline" ]]; then
  sweep > "$BASELINE"
  echo "[✓] Baseline saved: $BASELINE ($(wc -l <"$BASELINE") hosts)"
elif [[ -f "$BASELINE" ]]; then
  sweep > "$CURRENT"
  echo "=== NEW HOSTS ==="
  comm -13 <(sort "$BASELINE") <(sort "$CURRENT") | grep . || echo "None"
  echo "=== MISSING HOSTS ==="
  comm -23 <(sort "$BASELINE") <(sort "$CURRENT") | grep . || echo "None"
else
  echo "No baseline — run with --baseline first"
fi

Script 04 — SIEM-style Log Analyzer
Multi-Source Log Parser with IOC Detection
Parses multiple log formats, extracts IOCs (IPs, domains, hashes), correlates against a threat intel feed, and produces a structured JSON report.
log_analyzer.py PYTHON 
#!/usr/bin/env python3
"""
log_analyzer.py — Multi-format log parser with IOC detection
Supports: auth.log, nginx access.log, syslog, JSON logs
"""

import re, json, sys, argparse, logging
from pathlib    import Path
from datetime   import datetime, timezone
from collections import defaultdict, Counter
from ipaddress  import ip_address, ip_network, AddressValueError

log = logging.getLogger(__name__)

# ── PATTERNS ──────────────────────────────────────────
PATTERNS = {
  'ipv4':   re.compile(
    r'\b(?:(?:25[0-5]|2[0-4]\d|[01]?\d\d?)\.){3}(?:25[0-5]|2[0-4]\d|[01]?\d\d?)\b'
  ),
  'domain': re.compile(
    r'\b(?:[a-zA-Z0-9](?:[a-zA-Z0-9\-]{0,61}[a-zA-Z0-9])?\.)+[a-zA-Z]{2,}\b'
  ),
  'hash_sha256': re.compile(r'\b[0-9a-fA-F]{64}\b'),
  'hash_md5':    re.compile(r'\b[0-9a-fA-F]{32}\b'),
}

# Private ranges — skip these from IOC reports
PRIVATE_RANGES = [
  ip_network('10.0.0.0/8'), ip_network('172.16.0.0/12'),
  ip_network('192.168.0.0/16'), ip_network('127.0.0.0/8'),
]

class LogAnalyzer:
  def __init__(self, threat_feed: set = None):
    self.events      = []
    self.iocs        = defaultdict(set)
    self.ip_counts   = Counter()
    self.threat_hits = []
    self.threat_feed = threat_feed or set()

  def is_private(self, ip_str: str) -> bool:
    try:
      addr = ip_address(ip_str)
      return any(addr in net for net in PRIVATE_RANGES)
    except AddressValueError:
      return False

  def extract_iocs(self, line: str) -> dict:
    found = {}
    for ioc_type, pattern in PATTERNS.items():
      matches = pattern.findall(line)
      if ioc_type == 'ipv4':
        matches = [m for m in matches if not self.is_private(m)]
      if matches:
        found[ioc_type] = list(set(matches))
        self.iocs[ioc_type].update(matches)
    return found

  def analyze_line(self, line: str, source: str) -> None:
    iocs = self.extract_iocs(line)
    if not iocs:
      return
    event = {'source': source, 'raw': line.strip()[:300], 'iocs': iocs}

    # Threat intel correlation
    for ip in iocs.get('ipv4', []):
      self.ip_counts[ip] += 1
      if ip in self.threat_feed:
        event['threat_match'] = ip
        self.threat_hits.append(event)
        log.warning(f"[THREAT MATCH] {ip} in threat feed — line: {line[:80]}")
    self.events.append(event)

  def process_file(self, path: Path) -> None:
    log.info(f"Processing: {path}")
    with open(path, 'r', errors='replace') as fh:
      for line in fh:
        self.analyze_line(line, str(path))

  def report(self) -> dict:
    return {
      'generated_at': datetime.now(timezone.utc).isoformat(),
      'summary': {
        'total_events':   len(self.events),
        'unique_ips':      len(self.iocs.get('ipv4', [])),
        'threat_hits':    len(self.threat_hits),
      },
      'top_ips':      self.ip_counts.most_common(20),
      'threat_events': self.threat_hits,
      'all_iocs':      {k: list(v) for k, v in self.iocs.items()},
    }

def main():
  p = argparse.ArgumentParser(description='Log analyzer with IOC detection')
  p.add_argument('logs', nargs='+', help='Log files to analyze')
  p.add_argument('--feed', help='Threat feed file (one IP per line)')
  p.add_argument('--out', default='report.json')
  args = p.parse_args()

  feed = set()
  if args.feed:
    feed = {ln.strip() for ln in Path(args.feed).read_text().splitlines() if ln.strip()}

  analyzer = LogAnalyzer(threat_feed=feed)
  for log_path in args.logs:
    analyzer.process_file(Path(log_path))

  report = analyzer.report()
  Path(args.out).write_text(json.dumps(report, indent=2))
  log.info(f"Report → {args.out} | Threats: {report['summary']['threat_hits']}")

if __name__ == '__main__':
  logging.basicConfig(level=logging.INFO)
  main()

Script 05 — Threat Intel Enrichment
VirusTotal API — Bulk IOC Enrichment
Takes a list of file hashes, IPs, or domains, queries the VirusTotal API with rate limiting, and enriches each with verdicts, detection ratios, and tags.
vt_enrich.py PYTHON 
#!/usr/bin/env python3
"""
vt_enrich.py — VirusTotal bulk IOC enrichment
Rate-limited, retry-safe, outputs enriched JSON
"""
import time, json, sys, os
import requests
from pathlib import Path

VT_BASE    = "https://www.virustotal.com/api/v3"
API_KEY    = os.environ.get("VT_API_KEY", "")  # NEVER hardcode keys
RATE_LIMIT = 4   # free tier: 4 req/min
DELAY      = 60 / RATE_LIMIT

def vt_lookup(ioc: str, ioc_type: str) -> dict:
  """Query VT for hash|ip|domain with retry on 429."""
  endpoints = {
    'hash':   f"/files/{ioc}",
    'ip':     f"/ip_addresses/{ioc}",
    'domain': f"/domains/{ioc}",
  }
  url = VT_BASE + endpoints[ioc_type]
  headers = {"x-apikey": API_KEY}

  for attempt in range(3):
    resp = requests.get(url, headers=headers, timeout=15)
    if resp.status_code == 200:
      data = resp.json().get("data", {}).get("attributes", {})
      stats = data.get("last_analysis_stats", {})
      return {
        "ioc":      ioc,
        "type":     ioc_type,
        "malicious":  stats.get("malicious", 0),
        "suspicious": stats.get("suspicious", 0),
        "undetected": stats.get("undetected", 0),
        "tags":     data.get("tags", []),
        "verdict":  "MALICIOUS" if stats.get("malicious", 0) > 2 else "CLEAN",
      }
    elif resp.status_code == 429:
      wait = 60 * (attempt + 1)
      print(f"Rate limited — waiting {wait}s...", file=sys.stderr)
      time.sleep(wait)
    elif resp.status_code == 404:
      return {"ioc": ioc, "verdict": "NOT_FOUND"}
    else:
      return {"ioc": ioc, "error": resp.status_code}
  return {"ioc": ioc, "error": "max_retries"}

def enrich_list(iocs: list, ioc_type: str) -> list:
  results = []
  for i, ioc in enumerate(iocs):
    print(f"[{i+1}/{len(iocs)}] Querying: {ioc}")
    result = vt_lookup(ioc.strip(), ioc_type)
    results.append(result)
    if result.get("verdict") == "MALICIOUS":
      print(f"  ⚠ MALICIOUS — {result['malicious']} engines")
    time.sleep(DELAY)  # respect rate limit
  return results

if __name__ == "__main__":
  if not API_KEY:
    sys.exit("Set VT_API_KEY environment variable")
  # Example: pipe hashes from file
  hashes = sys.stdin.read().splitlines()
  results = enrich_list(hashes, "hash")
  Path("vt_results.json").write_text(json.dumps(results, indent=2))

Script 06 — Threat Hunting
Windows Threat Hunting Script
Comprehensive host triage — unsigned processes, suspicious parent-child chains, persistence mechanisms, network connections. Outputs structured report.
Invoke-ThreatHunt.ps1 PWSH 
#Requires -RunAsAdministrator
<# .SYNOPSIS
   Invoke-ThreatHunt.ps1 — Host triage & threat hunting
   Run from an elevated PowerShell session in a sandbox
#>

Set-StrictMode -Version Latest
$ErrorActionPreference = 'Continue'
$report = [ordered]@{ Timestamp = (Get-Date -Format o); Findings = @() }

function Add-Finding([string]$Category, [string]$Severity, [string]$Detail, $Data) {
  $report.Findings += [PSCustomObject]@{
    Category = $Category; Severity = $Severity
    Detail   = $Detail;   Data     = $Data
  }
  $color = switch ($Severity) { 'HIGH' {'Red'} 'MED' {'Yellow'} default {'Cyan'}}
  Write-Host "[$Severity] $Category → $Detail" -ForegroundColor $color
}

# ── 1. UNSIGNED PROCESSES ─────────────────────────
Write-Host "`n[*] Checking process signatures..."
Get-Process | ForEach-Object {
  try {
    $path = $_.MainModule.FileName
    if ($null -eq $path) { return }
    $sig = Get-AuthenticodeSignature $path
    if ($sig.Status -ne 'Valid') {
      Add-Finding 'UNSIGNED_PROC' 'HIGH' "$($_.Name) PID=$($_.Id)" @{
        Path      = $path
        SigStatus = $sig.Status
        MD5       = (Get-FileHash $path -Algorithm MD5).Hash
      }
    }
  } catch { # Access denied for system procs — expected }
}

# ── 2. SUSPICIOUS PARENT-CHILD CHAINS ────────────
Write-Host "`n[*] Checking process lineage..."
$suspicious = @(
  @{Parent='winword.exe';  Child='cmd.exe'}
  @{Parent='winword.exe';  Child='powershell.exe'}
  @{Parent='excel.exe';    Child='wscript.exe'}
  @{Parent='outlook.exe';  Child='cmd.exe'}
  @{Parent='explorer.exe'; Child='svchost.exe'}
)
$procs = Get-CimInstance Win32_Process | Select-Object Name, ProcessId, ParentProcessId, CommandLine
foreach ($rule in $suspicious) {
  $parents = $procs | Where-Object { $_.Name -eq $rule.Parent }
  foreach ($parent in $parents) {
    $children = $procs | Where-Object { $_.ParentProcessId -eq $parent.ProcessId -and $_.Name -eq $rule.Child }
    foreach ($child in $children) {
      Add-Finding 'PROC_CHAIN' 'HIGH' "$($rule.Parent) → $($rule.Child)" $child.CommandLine
    }
  }
}

# ── 3. PERSISTENCE — REGISTRY RUN KEYS ───────────
Write-Host "`n[*] Checking persistence mechanisms..."
$runKeys = @(
  'HKCU:\Software\Microsoft\Windows\CurrentVersion\Run'
  'HKLM:\Software\Microsoft\Windows\CurrentVersion\Run'
  'HKLM:\Software\Microsoft\Windows\CurrentVersion\RunOnce'
)
foreach ($key in $runKeys) {
  if (Test-Path $key) {
    Get-ItemProperty $key | Select-Object -Property * -ExcludeProperty PS* |
      Get-Member -MemberType NoteProperty | ForEach-Object {
        $val = (Get-ItemProperty $key).($_.Name)
        Add-Finding 'PERSISTENCE' 'MED' "RunKey: $($_.Name)" @{Key=$key; Value=$val}
      }
  }
}

# ── 4. ACTIVE NETWORK CONNECTIONS ─────────────────
Write-Host "`n[*] Checking network connections..."
Get-NetTCPConnection -State Established | ForEach-Object {
  $conn = $_
  $proc = Get-Process -Id $conn.OwningProcess -ErrorAction SilentlyContinue
  # Flag non-browser processes with outbound connections
  if ($proc.Name -notin @('chrome','firefox','msedge','svchost')) {
    Add-Finding 'NETWORK' 'MED' "$($proc.Name) → $($conn.RemoteAddress):$($conn.RemotePort)" $null
  }
}

# ── OUTPUT ────────────────────────────────────────
$outFile = "ThreatHunt_$(Get-Date -Format yyyyMMdd_HHmmss).json"
$report | ConvertTo-Json -Depth 5 | Out-File $outFile
Write-Host "`n[✓] Report: $outFile | Findings: $($report.Findings.Count)" -ForegroundColor Green

Script 07 — Event Log Analysis
Security Event Log Hunter
Query Windows Security, System, and PowerShell event logs for high-value forensic event IDs — logon failures, privilege use, service installs, and lateral movement indicators.
Get-SecurityEvents.ps1 PWSH 
# Get-SecurityEvents.ps1 — Query high-value Windows event IDs
param(
  [DateTime]$StartTime = (Get-Date).AddHours(-24),
  [string]  $ComputerName = $env:COMPUTERNAME
)

# High-value Security Event IDs
$eventMap = [ordered]@{
  4624 = "Successful Logon"
  4625 = "Failed Logon"
  4648 = "Logon with Explicit Credentials"
  4672 = "Special Privileges Assigned"
  4698 = "Scheduled Task Created"
  4702 = "Scheduled Task Modified"
  4720 = "User Account Created"
  4732 = "Member Added to Security Group"
  7045 = "New Service Installed"
  4104 = "PowerShell Script Block"
}

$filterXml = @"
<QueryList>
  <Query Id="0">
    <Select Path="Security">
      *[System[EventID=4624 or EventID=4625 or EventID=4648 or
               EventID=4672 or EventID=4698 or EventID=4720 or
               EventID=4732]
        and System[TimeCreated[@SystemTime >= '$($StartTime.ToUniversalTime().ToString("o"))']]]
    </Select>
    <Select Path="System">
      *[System[EventID=7045]]
    </Select>
    <Select Path="Microsoft-Windows-PowerShell/Operational">
      *[System[EventID=4104]]
    </Select>
  </Query>
</QueryList>
"@

Write-Host "[*] Querying events since $StartTime on $ComputerName"

Get-WinEvent -FilterXml $filterXml -ComputerName $ComputerName -ErrorAction SilentlyContinue |
  Sort-Object TimeCreated |
  Select-Object TimeCreated, Id,
    @{n='Description';e={$eventMap[[int]$_.Id] ?? "Event $($_.Id)"}},
    Message |
  Format-Table -AutoSize |
  Out-Host

# Highlight failed logon bursts (brute force indicator)
$failedLogons = Get-WinEvent -FilterXml $filterXml -ErrorAction SilentlyContinue |
  Where-Object { $_.Id -eq 4625 }
if ($failedLogons.Count -gt 20) {
  Write-Host "[!] ALERT: $($failedLogons.Count) failed logons in last 24h" -ForegroundColor Red
}

Scripting Vulnerabilities & Defenses
Security scripts that are themselves insecure are ironic liabilities. Every pattern here is a real-world vulnerability class found in production security tooling.
Reality: Many production security scripts contain injection flaws, hardcoded credentials, race conditions, and TOCTOU bugs — creating attack surface within the blue team's own toolchain.
Vulnerability 01
Command Injection — Bash & Python
The most common scripting flaw. Unsanitized user input passed to a shell command or os.system() allows arbitrary command execution.
/* BASH — Command Injection */
−
user_input="$1" eval "ls $user_input" # INJECTABLE: ; rm -rf /
−

Script 08 — Linux IR Triage
Linux First-Responder Collection Script
Collects volatile evidence in order of volatility (memory state → running processes → network → disk artifacts). Designed to run within the first 5 minutes of a Linux incident.
linux_triage.sh BASH 
#!/usr/bin/env bash
# linux_triage.sh — First-responder evidence collection
# MUST run as root. Collects volatile evidence before shutdown.
# Output: timestamped tarball of all artifacts

set -euo pipefail
[[ $EUID -ne 0 ]] && { echo "Run as root"; exit 1; }

TS=$(date -u +%Y%m%d_%H%M%SZ)
HOST=$(hostname -s)
OUT="/tmp/triage_${HOST}_${TS}"
mkdir -p "$OUT"/{volatile,network,persist,logs}
trap "tar -czf ${OUT}.tar.gz -C /tmp triage_${HOST}_${TS} && rm -rf '$OUT' && echo '[✓] Package: ${OUT}.tar.gz'" EXIT

log() { echo "[$(date -u +%T)] $*"; }
cap() {
  # cap label command → saves output to file, handles errors
  local label="$1"; shift
  log "Collecting: $label"
  "$@" > "$OUT/${label}.txt" 2>&1 || echo "[WARN] $label failed"
}

# ── PHASE 1: MOST VOLATILE FIRST ─────────────────
log "=== PHASE 1: VOLATILE DATA ==="

cap volatile/who            who -a

cap volatile/w              w

cap network/routes          ip route show

Script 09 — Windows IR Containment
Windows Network Isolation + Evidence Collection
Blocks all non-essential outbound traffic via Windows Firewall, preserves volatile state, and prepares a handoff package for forensic analysis — all reversible.
Invoke-IRContain.ps1 PWSH 
#Requires -RunAsAdministrator
<#
.SYNOPSIS  Windows IR containment + volatile evidence collection
.NOTES     -WhatIf supported for dry-run; -Undo reverses isolation
#>
param([switch]$Undo, [switch]$SkipIsolation)

$TS      = Get-Date -Format "yyyyMMdd_HHmmssZ"
$Host_   = $env:COMPUTERNAME
$OutDir  = "C:\IR_${Host_}_${TS}"
New-Item -Path $OutDir -ItemType Directory -Force | Out-Null

function Save($name, $data) {
  $data | Out-File -FilePath "$OutDir\$name.txt" -Encoding UTF8
}

# ── VOLATILE EVIDENCE ─────────────────────────────
Write-Host "[*] Collecting volatile state..." -ForegroundColor Cyan

Save "01_datetime"    (Get-Date -Format o)
Save "02_processes"   (Get-CimInstance Win32_Process |
  Select-Object Name, ProcessId, ParentProcessId,
    CommandLine, ExecutablePath,
    @{n='StartTime';e={$_.CreationDate}} |
  Format-List)
Save "03_network"     (Get-NetTCPConnection | Select-Object LocalAddress, LocalPort,
  RemoteAddress, RemotePort, State, OwningProcess | Format-Table)
Save "04_loggedon"    (Get-CimInstance Win32_LoggedOnUser)
Save "05_services"    (Get-Service | Where-Object {$_.Status -eq 'Running'})
Save "06_startup"     (Get-CimInstance Win32_StartupCommand)
Save "07_shares"      (Get-SmbShare)
Save "08_scheduled"   (Get-ScheduledTask | Where-Object State -ne 'Disabled')
Save "09_run_keys"    (Get-ItemProperty 'HKCU:\Software\Microsoft\Windows\CurrentVersion\Run')
Save "10_prefetch"    (Get-ChildItem C:\Windows\Prefetch | Sort-Object LastWriteTime -Descending | Select-Object -First 50)

# ── NETWORK ISOLATION ─────────────────────────────
if (-not $SkipIsolation -and -not $Undo) {
  Write-Host "[!] Isolating host — blocking all outbound except RDP (3389)" -ForegroundColor Red
  # Block all outbound
  New-NetFirewallRule -DisplayName "IR_BLOCK_OUTBOUND" `
    -Direction Outbound -Action Block -Profile Any `
    -Description "IR containment $(Get-Date -Format o)" | Out-Null
  # Allow RDP so responders can still connect
  New-NetFirewallRule -DisplayName "IR_ALLOW_RDP" `
    -Direction Outbound -Action Allow -Protocol TCP -RemotePort 3389 | Out-Null
  Write-Host "[✓] Host isolated. Run with -Undo to restore." -ForegroundColor Yellow
}

if ($Undo) {
  Remove-NetFirewallRule -DisplayName "IR_BLOCK_OUTBOUND" -ErrorAction SilentlyContinue
  Remove-NetFirewallRule -DisplayName "IR_ALLOW_RDP"    -ErrorAction SilentlyContinue
  Write-Host "[✓] Isolation rules removed — network restored" -ForegroundColor Green
  exit 0
}

# ── PACKAGE ───────────────────────────────────────
Compress-Archive -Path $OutDir -DestinationPath "${OutDir}.zip"
Write-Host "[✓] Evidence package: ${OutDir}.zip" -ForegroundColor Green
(Get-FileHash "${OutDir}.zip").Hash | Out-File "${OutDir}.zip.sha256"
Write-Host "[✓] Hash: $(Get-Content "${OutDir}.zip.sha256")"
